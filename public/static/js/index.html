验证插件使用说明

1、页面中引入jquery.js库(版本>=1.4.4)与coo.js

2、加入JS脚本：$(function(){$(form表单).checkform()});

3、验证标签对象使用类名class属性，验证标签以"-"符号分隔。例如：class="checkform-need-email"：必填字段，且只能输入邮箱，也可添加其他类名class="myClass checkform-need-email"，即可使用多个类名，其实need标签已可忽略，因为email验证标签已包含不能为空的要求，验证顺序按验证标签顺序执行，还有一种是扩展需验证的expr，更多验证标签与详细教程请参考下面的说明

4、自定义验证规则，使用checkform-reg自定义属性，例如：class="checkform-reg" checkform-reg="^\d{4}$"，验证不通过时提示：“格式错误”

5、使用checkform-error-class自定义属性，自定义被验证对象的验证不通过样式，验证不通过时被验证对象将添加该样式，通过时自动去除

6、验证提示的位置，使用checkform-label-position自定义属性，值有self、top、bottom、left、right、end、parent、absleft、absright、expr、(num num)，end：在被验证对象的父元素的内部最后，parent：被验证对象的父元素后面，num：数字(漂浮，可负数，分别对应以被验证对象左上角为0开始的left、top)，[absleft|absright]：漂浮在被验证对象的左/右侧，expr：css选择器(当不是之前所有的值时即当作css选择器执行，提示将显示在该expr对象的前面)，不设置该自定义属性或为空或right即默认在被验证对象的后面(被验证对象为checkbox或radio时为end)

7、自定义提示，使用checkform-label-success自定义属性即在验证通过时显示出来，使用checkform-label-error自定义属性即在验证不通过时提示error的信息。例如：class="checkform-need" checkform-label-error="干嘛不输入呢？"，验证不通过时将会提示："干嘛不输入呢？"，而不是默认的"此项必填"，每种自定义信息都可以设定各自的css与area，例如checkform-label-error-class、checkform-label-area，使用checkform-label-area自定义属性时checkform-label-position将会无效，checkform-label-area的值为存放验证失败提示的容器的expr，class="checkform-need-num" 可使用自定义属性 need-label-error="必填" num-label-error="必须数字" 设定各个验证标签的提示信息(增加对应的前缀)(包括可设定各自的label-error-class、label-area，例如num-label-error-class、num-label-area)

8、修改提示样式，在样式表中给label.error定义样式即可，或使用checkform-label-error-class自定义属性指定一个class样式名称(success与error一样规则)

	let options = {
		debug: false, //调试模式(不提交表单)
		tag: 'checkform', //验证标识前缀
		overlayText: '', //提交后表单添加遮罩的文字, 不设置即不添加
		overlay: '', //提交后表单添加遮罩的背景颜色
		overload: false, //提示信息使用overloadError
		addEles: null, //扩展验证控件(没有使用tag前缀类名的控件), 格式 { '.required': /.+/, '[diy="img"]': /\.(jpg|jpeg|png|gif|bmp)$/ }
		labelNon: false, //不使用label提示信息
		labelPosition: '', //提示信息位置, 有效值为 self|top|bottom|left|right|end|parent|absleft|absright|expr|(x y)
		errorClass: null, //验证不通过添加的提示样式名称, this验证控件
		reset: null, //表单重置后执行, this表单
		before: null, //表单提交时验证前执行, 可返回boolean, true不验证直接提交, false终止提交, this表单
		callback: null //表单提交时验证通过后执行, 返回false终止提交, this表单
	}
	$(expr).checkform(options);

		if(!$('#infloat').checkform('!float')){
			$('#infloat').focus();
			alert('该项目可不填, 但必须为数字(可带小数)');
			return false;
		}
		if(!$('#innum').checkform('num')){

上传控件

	$(expr).ajaxupload({
		url: '/upload', //上传提交的目标网址
		name: 'filename', //非file控件上传时指定的提交控件名称
		loading: '', //上传中在右边显示的图片路径
		innerFile: $.browser.mobile, //直接在内部生成input:file,为兼容某些浏览器不支持模拟点击
		fileType: ['jpg', 'jpeg', 'png', 'gif', 'bmp', 'svg'], //允许上传文件类型,后缀名,数组或字符串(逗号隔开)
		dataType: 'json', //请求类型
		data: null, //上传时一同提交的数据
		rightnow: false, //不需要经过change之类的操作,立刻上传(控件已选择了文件)
		target: '', //表单提交目标
		multiple: false, //多文件选择(只支持HTML5浏览器)
		beforeSelect: null, //选择前执行, 非file控件有效, 若返回false即终止选择
		before: null, //上传前执行, 若返回false即终止上传, 接受一个参数:选择的文件数量
		cancel: null, //终止上传后执行
		uploading: null, //正在上传时执行
		callback: null, //上传操作完毕后返回的回调函数(函数(即success函数)|字面量对象)
		error: null, //上传操作失败后执行(即error函数,比callback.error优先)
		complete: null, //上传操作完毕后执行(即complete函数,比callback.complete优先)
		replace: null, //替换上传操作, 例如使用又拍云替换ajaxupload的默认上传, 接受一个参数:file控件
		debug: false //保留iframe,form
	});

HTML5 拖拽上传 (可多选)

支持HTML5的浏览器才有效, 否则file类型上传控件将自动使用 ajaxupload 脚本, 其他拖拽类型即无效

	$(expr).html5upload({
		url: '/upload', //文件数据处理URL
		name: 'filename', //上传字段名,如果字符串末尾加上叹号,即强制使用该字符串
		change: true, //选择文件后立即上传(不支持非file控件)
		rightnow: false, //不需要经过change之类的操作,立刻上传(控件已选择了文件)(不支持非file控件)
		data: null, //附加参数
		splitSize: 0, //把文件切片上传,例如切为5M:5*1024*1000,0为不切片(后端需自行处理切片组合,自动增加附加参数split_name:options.name+'_'+index,split_total:总数)
		fileType: ['jpg', 'jpeg', 'png', 'gif', 'bmp', 'svg'], //允许上传文件类型,后缀名,数组或字符串(逗号隔开)
		dataType: 'json', //请求类型
		dragover: null, //拖动到对象上时执行
		dragleave: null, //拖动到对象外面时执行
		select: null, //选择文件后执行
		before: null, //上传之前执行,返回false即取消上传
		progress: null, //数据正在被post的过程中周期性执行,接受三个参数 e, e.loaded, e.total
		success: null, //post操作成功时执行
		error: null //失败时执行,例如选择了不允许的文件类型
	});
